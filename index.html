<script>

function calculateResults() {

    const form = document.forms["diagForm"];

    let total = 0;

    let count = 0;

 

    for (let i = 1; i <= 15; i++) {

        const val = Number(form["q"+i].value);

        total += val;

        count++;

    }

 

    const globalScore = total / count;

    let level = "";

    if (globalScore <= 2.5) level = "Débutant";

    else if (globalScore <= 3.8) level = "Moyen";

    else level = "Avancé";

 

    document.getElementById("results").innerHTML = `

        <div class="result-card">

            <h2>Résultat global : ${globalScore.toFixed(2)}/5</h2>

            <p><strong>Niveau :</strong> ${level}</p>

 

            <div class="gauge">

                <div class="gauge-fill" style="width:${globalScore * 20}%;"></div>

            </div>

        </div>

    `;

}

</script>
